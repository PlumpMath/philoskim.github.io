3. 소스코드 예제 표기요령

(defn get-name []
  (println "Enter Your Name:")
  (let [name (read-line)]
    (println "Hello," name)
    name))

(get-name)
; >> Enter Your Name:
; << Mr. Kim
; >> Hello, Mr. Kim
; => "Mr. Kim"
